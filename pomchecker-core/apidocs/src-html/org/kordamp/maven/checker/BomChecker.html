018package org.kordamp.maven.checker;
019
020import org.apache.maven.model.Dependency;
021import org.apache.maven.model.Model;
022import org.apache.maven.project.MavenProject;
023
024import java.util.ArrayList;
025import java.util.List;
026
027/**
028 * Checks if a POM file is a minimal BOM file.
029 *
030 * The following blocks are required:
031 * <ul>
032 *   <li>&lt;dependencyManagement&gt;</li>
033 * </ul>
034 *
035 * The following blocks are forbidden:
036 * <ul>
037 *   <li>&lt;build&gt;</li>
038 *   <li>&lt;reporting&gt;</li>
039 *   <li>&lt;dependencies&gt;</li>
040 *   <li>&lt;repositories&gt;</li>
041 *   <li>&lt;pluginRepositories&gt;</li>
042 *   <li>&lt;profiles&gt;</li>
043 *   <li>&lt;modules&gt;</li>
044 * </ul>
045 *
046 * @author Andres Almiray
047 * @since 1.0.0
048 */
049public class BomChecker {
050    /**
051     * Checks the resolved model of the given MaveProject for compliance.
052     *
053     * @param log     the logger to use.
054     * @param project the project to be checked.
055     * @throws PomCheckException if the POM is invalid
056     */
057    public static void check(Logger log, MavenProject project) throws PomCheckException {
058        Model model = project.getOriginalModel();
059
060        List<String> errors = new ArrayList<>();
061
062        // 1. is it packaged as 'pom'?
063        log.debug("Checking <packaging>");
064        if (!"pom".equals(model.getPackaging())) {
065            errors.add("The value of <packaging> must be 'pom'.");
066        }
067
068        log.debug("Checking <dependencyManagement>");
069        // 2. must have a <dependencyManagement> block
070        if (null != model.getDependencyManagement()) {
071            List<Dependency> dependencies = model.getDependencyManagement().getDependencies();
072            if (dependencies == null || dependencies.isEmpty()) {
073                errors.add("No dependencies have been defined in <dependencyManagement>.");
074            }
075        } else {
076            errors.add("No <dependencyManagement> block has been defined.");
077        }
078
079        log.debug("Checking <build>");
080        if (null != model.getBuild()) {
081            errors.add("The <build> block should not be present.");
082        }
083
084        log.debug("Checking <reporting>");
085        if (null != model.getReporting()) {
086            errors.add("The <reporting> block should not be present.");
087        }
088
089        log.debug("Checking <dependencies>");
090        if (null != model.getDependencies() && !model.getDependencies().isEmpty()) {
091            errors.add("The <dependencies> block should not be present.");
092        }
093
094        log.debug("Checking <repositories>");
095        if (null != model.getRepositories() && !model.getRepositories().isEmpty()) {
096            errors.add("The <repositories> block should not be present.");
097        }
098
099        log.debug("Checking <pluginRepositories>");
100        if (null != model.getPluginRepositories() && !model.getPluginRepositories().isEmpty()) {
101            errors.add("The <pluginRepositories> block should not be present.");
102        }
103
104        log.debug("Checking <profiles>");
105        if (null != model.getProfiles() && !model.getProfiles().isEmpty()) {
106            errors.add("The <profiles> block should not be present.");
107        }
108
109        log.debug("Checking <modules>");
110        if (null != model.getModules() && !model.getModules().isEmpty()) {
111            errors.add("The <modules> block should not be present.");
112        }
113
114        if (!errors.isEmpty()) {
115            StringBuilder b = new StringBuilder("\nThe POM file\n")
116                .append(project.getFile().getAbsolutePath())
117                .append("\nis not a valid BOM due to the following reasons:\n");
118            for (String s : errors) {
119                b.append(" * ").append(s).append("\n");
120            }
121
122            throw new PomCheckException(b.toString());
123        } else {
124            log.info("BOM {} passes all checks.", project.getFile().getAbsolutePath());
125        }
126    }
127}
