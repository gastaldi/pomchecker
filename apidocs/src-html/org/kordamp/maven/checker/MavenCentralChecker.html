028import java.io.FileReader;
029import java.util.ArrayList;
030import java.util.List;

032import static org.apache.commons.lang3.StringUtils.isBlank;
033import static org.apache.commons.lang3.StringUtils.isNotBlank;

035/**
036 * Checks if a POM complies with the rules for uploading to Maven Central.
037 *
038 * The following blocks are required:
039 * <ul>
040 *   <li>&lt;groupId&gt;</li>
041 *   <li>&lt;artifactId&gt;</li>
042 *   <li>&lt;version&gt;</li>
043 *   <li>&lt;name&gt;</li>
044 *   <li>&lt;description&gt;</li>
045 *   <li>&lt;url&gt;</li>
046 *   <li>&lt;licenses&gt;</li>
047 *   <li>&lt;scm&gt;</li>
048 * </ul>
049 *
050 * All previous blocks may be supplied by a parent POM with the exception of &lt;artifactId&gt;.
051 *
052 * The following blocks are forbidden if {@code strict = true}
053 * <ul>
054 *   <li>&lt;repositories&gt;</li>
055 *   <li>&lt;pluginRepositories&gt;</li>
056 * </ul>
057 *
058 * @see <a href="http://maven.apache.org/repository/guide-central-repository-upload.html">http://maven.apache.org/repository/guide-central-repository-upload.html</a>
059 * @see <a href="https://central.sonatype.org/pages/requirements.html">https://central.sonatype.org/pages/requirements.html</a>
060 *
061 * @author Andres Almiray
062 * @since 1.0.0
063 */
064public class MavenCentralChecker {
065    /**
066     * Checks the resolved model of the given MaveProject for compliance.
067     *
068     * @param log     the logger to use.
069     * @param project the project to be checked.
070     * @param release if {@code true} checks if version is not -SNAPSHOT.
071     * @param strict  if {@code true} checks that &lt;repositories&gt; and &lt;pluginRepositories&gt; are not present
072     * @throws PomCheckException if the POM is invalid
073     */
074    public static void check(Logger log, MavenProject project, boolean release, boolean strict) throws PomCheckException {
075        Model fullModel = project.getModel();
076        Model originalModel = project.getOriginalModel();
077
078        List<String> errors = new ArrayList<>();
079
080        // sanity checks. redundant?
081        if (isBlank(fullModel.getGroupId())) {
082            errors.add("<groupId> can not be blank.");
083        }
084        if (isBlank(fullModel.getArtifactId())) {
085            errors.add("<artifactId> can not be blank.");
086        }
087        if (isBlank(fullModel.getVersion())) {
088            errors.add("<version> can not be blank.");
089        }
090
091        if (isBlank(fullModel.getName())) {
092            String parentName = resolveParentName(project.getFile().getParentFile(), fullModel);
093            if (isBlank(parentName)) {
094                errors.add("<name> can not be blank.");
095            } else {
096                log.warn("<name> is not defined in POM. Will use value from parent: " +
097                    System.lineSeparator() + "\t" + parentName);
098            }
099        }
100
101        if (isBlank(fullModel.getDescription())) {
102            errors.add("<description> can not be blank.");
103        }
104        if (isBlank(originalModel.getDescription())) {
105            log.warn("<description> is not defined in POM. Will use value from parent: " +
106                System.lineSeparator() + "\t" + fullModel.getDescription());
107        }
108
109        if (isBlank(fullModel.getUrl())) {
110            errors.add("<url> can not be blank.");
111        }
112        if (isBlank(originalModel.getUrl())) {
113            log.warn("<url> is not defined in POM. Will use computed value from parent: " +
114                System.lineSeparator() + "\t" + fullModel.getUrl());
115        }
116
117        if (release && fullModel.getVersion().endsWith("-SNAPSHOT")) {
118            errors.add("<version> can not be -SNAPSHOT.");
119        }
120
121        if (fullModel.getLicenses() != null) {
122            if (!fullModel.getLicenses().isEmpty()) {
123                // verify that all licenses have <name> & <url>
124                for (int i = 0; i < fullModel.getLicenses().size(); i++) {
125                    License license = fullModel.getLicenses().get(i);
126                    if (isBlank(license.getName()) && isBlank(license.getUrl())) {
127                        errors.add("License " + i + " must define <name> and <url>.");
128                    }
129                }
130            } else {
131                errors.add("<licenses> block is required but was left empty.");
132            }
133        } else {
134            errors.add("<licenses> block is required but was left undefined.");
135        }
136
137        if (fullModel.getDevelopers() != null) {
138            if (!fullModel.getDevelopers().isEmpty()) {
139                // verify that all developers have at least one of [id, name, organization, organizationUrl]
140                for (int i = 0; i < fullModel.getDevelopers().size(); i++) {
141                    Developer developer = fullModel.getDevelopers().get(i);
142                    if (isBlank(developer.getId()) &&
143                        isBlank(developer.getName()) &&
144                        isBlank(developer.getOrganization()) &&
145                        isBlank(developer.getOrganizationUrl())) {
146                        errors.add("Developer " + i + " must define at least one of <id>, <name>, <organization>, <organizationUrl>.");
147                    }
148                }
149            } else {
150                errors.add("<developers> block is required but was left empty.");
151            }
152        } else {
153            errors.add("<developers> block is required but was left undefined.");
154        }
155
156        if (fullModel.getScm() == null) {
157            errors.add("The <scm> block is required.");
158        }
159
160        if (null != originalModel.getRepositories() && !originalModel.getRepositories().isEmpty()) {
161            if (strict) {
162                errors.add("The <repositories> block should not be present.");
163            } else {
164                log.warn("The <repositories> block should not be present.");
165            }
166        }
167        if (null != originalModel.getPluginRepositories() && !originalModel.getPluginRepositories().isEmpty()) {
168            if (strict) {
169                errors.add("The <pluginRepositories> block should not be present.");
170            } else {
171                log.warn("The <pluginRepositories> block should not be present.");
172            }
173        }
174
175        if (!errors.isEmpty()) {
176            StringBuilder b = new StringBuilder("\nThe POM file\n")
177                .append(project.getFile().getAbsolutePath())
178                .append("\ncannot be uploaded to Maven Central due to the following reasons:\n");
179            for (String s : errors) {
180                b.append(" * ").append(s).append("\n");
181            }
182
183            throw new PomCheckException(b.toString());
184        }
185    }
186
187    private static String resolveParentName(File directory, Model fullModel) {
188        Parent parent = fullModel.getParent();
189
190        while (parent != null) {
191            if (isNotBlank(parent.getRelativePath())) {
192                File pomFile = new File(directory, parent.getRelativePath());
193                if (!pomFile.getName().endsWith("pom.xml")) {
194                    pomFile = new File(pomFile, "pom.xml");
195                }
196
197                if (pomFile.exists()) {
198                    MavenProject parentProject = readProject(pomFile);
199                    Model parentModel = parentProject.getModel();
200                    if (isNotBlank(parentModel.getName())) {
201                        return parentModel.getName();
202                    } else {
203                        directory = pomFile.getParentFile();
204                        parent = parentModel.getParent();
205                    }
206                } else {
207                    // parent should be available from a repository
208                    // TODO: resolve parent
209                    return null;
210                }
211            } else {
212                // check 1 level up
213                File pomFile = new File(directory, "../pom.xml");
214                if (pomFile.exists()) {
215                    MavenProject parentProject = readProject(pomFile);
216                    Model parentModel = parentProject.getModel();
217                    if (isNotBlank(parentModel.getName())) {
218                        return parentModel.getName();
219                    } else {
220                        directory = pomFile.getParentFile();
221                        parent = parentModel.getParent();
222                    }
223                } else {
224                    // parent should be available from a repository
225                    // TODO: resolve parent
226                    return null;
227                }
228            }
229        }
230
231        return null;
232    }
233
234    private static MavenProject readProject(File pom) {
235        try {
236            FileReader reader = new FileReader(pom);
237            MavenXpp3Reader mavenReader = new MavenXpp3Reader();
238            return new MavenProject(mavenReader.read(reader));
239        } catch (Exception e) {
240            throw new IllegalArgumentException(e);
241        }
242    }
