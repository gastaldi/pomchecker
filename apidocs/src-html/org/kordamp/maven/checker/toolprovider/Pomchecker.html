018package org.kordamp.maven.checker.toolprovider;
019
020import org.kordamp.jipsy.annotations.ServiceProviderFor;
021import org.kordamp.maven.checker.cli.Main;
022
023import java.io.PrintWriter;
024import java.util.spi.ToolProvider;
025
026/**
027 * @author Andres Almiray
028 * @since 1.1.0
029 */
030@ServiceProviderFor(ToolProvider.class)
031public class Pomchecker implements ToolProvider {
032    @Override
033    public String name() {
034        return "pomchecker";
035    }
036
037    @Override
038    public int run(PrintWriter out, PrintWriter err, String... args) {
039        return Main.run(out, err, args);
040    }
041
042    public static void main(String[] args) {
043        PrintWriter out = new PrintWriter(System.out, true);
044        PrintWriter err = new PrintWriter(System.err, true);
045        System.exit(Main.run(out, err, args));
046    }
047}
